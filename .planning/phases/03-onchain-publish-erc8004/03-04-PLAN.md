    ```typescript
    export async function publishRatingFor(subject: SubjectId): Promise<{ cid: string; reasoningHash: Hex; txHash: Hex }> {
      const { doc, reasoningHash } = await rate(subject);          // Phase 2, ONCE
      const canonical = canonicalizeDoc(doc);                      // EXACT hashed bytes (Pitfall 1)
      const cid = await pin(canonical);                            // bare CID
      let hash;
      try {
        hash = await walletClient.writeContract({
          address: REGISTRY, abi: ratingRegistryAbi, functionName: "publishRating",
          args: [doc.subject.address, doc.grade.uint8, reasoningHash, doc.confidence, cid],
          account, chain: mantle,
        });
      } catch (e) { throw redactRpcError(e); }                     // reuse rpc.ts scrub
      const receipt = await publicClient.waitForTransactionReceipt({ hash });
      const [evt] = parseEventLogs({ abi: ratingRegistryAbi, eventName: "RatingPublished", logs: receipt.logs });
      if (evt.args.reasoningHash !== reasoningHash || evt.args.cid !== cid)
        throw new Error("on-chain RatingPublished diverged from intended hash/cid");
      return { cid, reasoningHash, txHash: hash };
    }
    ```
